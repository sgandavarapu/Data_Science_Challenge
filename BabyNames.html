def popincdec(sumnameyear):
    #Additional processing for Popularity Inc/Dec
    #Function that defines increase and decrease in popularity for each name across years.
    df={}
    final_inc={}
    final_dec={}
    final_maxcount={}
    
    sorted_sumnameyear = sorted(sumnameyear.items(), key=operator.itemgetter(0), reverse=True) #sort by name
    oldname =  'null'
    max_count =0.001
    for key,values in sorted_sumnameyear:
        #print key
        #print values
        parts = key.strip().split(",") #name and year
        newname = parts[0] #person name
        df[parts[1]]=values #counts stored into year for each name
        if oldname<>newname:
            sorted_df = sorted(df.items(), key=operator.itemgetter(1),reverse=True) #sort by count value, 
            #print sorted_df
            for df_keys,df_values in sorted_df:
                max_year=int(df_keys) #find year in which max count turns up
                max_count=int(df_values)  # find max count
                break
            sorted_year_df = sorted(df.items(), key=operator.itemgetter(0)) #sort by year
            curr_min1=int(max_count) #initialize min before max
            curr_min2=int(max_count) #initialize min after max
            for df_keys,df_values in sorted_year_df: #implement logic for minimum bothe before and after maximum has been achieved
                if max_year>int(df_keys):
                    if curr_min1>df_values:
                        curr_min1=df_values
                if max_year<int(df_keys):
                    if curr_min2>df_values:
                        curr_min2=df_values
                perc_inc = round(100.0*(max_count-curr_min1)/float(curr_min1),3)
                perc_dec = round(100.0*(max_count-curr_min2)/float(max_count),3)
            #print oldname
            #print "max perc inc is " + str(perc_inc)
            #print "max perc dec is " + str(perc_dec)
            final_inc[oldname] = perc_inc
            final_dec[oldname] = perc_dec
            final_maxcount[oldname] = max_count
            
            df={} #initilize the dictionary with year as key and appearances as value for each name
            oldname = newname
        else:
            df[parts[1]]=values
            oldname = newname
    return final_inc,final_dec
